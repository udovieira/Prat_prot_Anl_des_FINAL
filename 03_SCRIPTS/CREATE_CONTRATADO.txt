CREATE TABLE `connectwork`.`CONTRATADO` (
    `ID_CONTRATADO` INT NOT NULL,
    `ID_ORCAMENTO` INT NOT NULL,
    `DS_NM_CONTRATADO` VARCHAR(250) NOT NULL,
    `DS_EMAIL` VARCHAR(250) NOT NULL,
    `DS_SENHA` VARCHAR(250) NOT NULL,
    `NR_TELEFONE` INT NOT NULL,
    `DS_TIPO_SERVICO` VARCHAR(250) NOT NULL,
    `DT_CADASTRO` DATETIME NOT NULL,
    PRIMARY KEY (`ID_CONTRATADO`),
    FOREIGN KEY (`ID_ORCAMENTO`)
        REFERENCES `connectwork`.`ORCAMENTO` (`ID_ORCAMENTO`)
);
