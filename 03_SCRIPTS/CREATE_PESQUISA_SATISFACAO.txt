CREATE TABLE `connectwork`.`PESQUISA_SATISFACAO` (
    `ID_SATISFACAO` INT NOT NULL,
    `ID_CONTRATANTE` INT NOT NULL,
    `DS_SATISFACAO_SERVICO` VARCHAR(250) NOT NULL,
    `NR_AVALIACAO_PRESTADOR` INT NOT NULL,
    `DS_SATISFACAO_PRESTADOR` VARCHAR(250) NOT NULL,
    PRIMARY KEY (`ID_SATISFACAO`),
    FOREIGN KEY (`ID_CONTRATANTE`)
        REFERENCES `connectwork`.`CONTRATANTE` (`ID_CONTRATANTE`)
);
