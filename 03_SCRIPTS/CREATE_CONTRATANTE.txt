CREATE TABLE `connectwork`.`CONTRATANTE` (
    `ID_CONTRATANTE` INT NOT NULL,
    `ID_SERVICO` INT NOT NULL,
    `DS_NM_CONTRATANTE` VARCHAR(250) NOT NULL,
    `DS_ENDERECO` VARCHAR(250) NOT NULL,
    `NR_ENDERECO` INT NOT NULL,
    `DS_BAIRO` VARCHAR(250) NOT NULL,
    `NR_CEP` INT NOT NULL,
    `DS_ESTADO` VARCHAR(250) NOT NULL,
    `DS_CIDADE` VARCHAR(250) NOT NULL,
    PRIMARY KEY (`ID_CONTRATANTE`),
    FOREIGN KEY (`ID_SERVICO`)
        REFERENCES `connectwork`.`SERVICO` (`ID_SERVICO`)
);
