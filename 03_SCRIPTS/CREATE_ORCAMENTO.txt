CREATE TABLE `connectwork`.`ORCAMENTO` (
    `ID_ORCAMENTO` INT NOT NULL,
    `ID_SERVICO` INT NOT NULL,
    `ID_CONTRATANTE` INT NOT NULL,
    `NR_VL_ORCAMENTO` FLOAT NOT NULL,
    `NR_VL_ORCAMENTO_ENTRADA` FLOAT NOT NULL,
    `DT_ORCAMENTO` DATETIME NOT NULL,
    `DS_STATUS_ORCAMENTO` VARCHAR(250) NOT NULL,
    PRIMARY KEY (`ID_ORCAMENTO`),
    FOREIGN KEY (`ID_SERVICO`)
        REFERENCES `connectwork`.`SERVICO` (`ID_SERVICO`),
    FOREIGN KEY (`ID_CONTRATANTE`)
        REFERENCES `connectwork`.`CONTRATANTE` (`ID_CONTRATANTE`)
);
