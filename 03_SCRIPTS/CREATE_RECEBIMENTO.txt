CREATE TABLE `connectwork`.`RECEBIMENTO` (
    `ID_RECEBIMENTO` INT NOT NULL,
    `ID_ORCAMENTO` INT NOT NULL,
    `ID_CONTRATANTE` INT NOT NULL,
    `ID_TIPO_PAGAMENTO` INT NOT NULL,
    `ID_DADOS_BANCARIOS` INT NOT NULL,
    `NR_VL_RECEBIMENTO` FLOAT NOT NULL,
    `DT_RECEBIMENTO` DATETIME NOT NULL,
    PRIMARY KEY (`ID_RECEBIMENTO`),
    FOREIGN KEY (`ID_ORCAMENTO`)
        REFERENCES `connectwork`.`ORCAMENTO` (`ID_ORCAMENTO`),
    FOREIGN KEY (`ID_CONTRATANTE`)
        REFERENCES `connectwork`.`CONTRATANTE` (`ID_CONTRATANTE`),
    FOREIGN KEY (`ID_TIPO_PAGAMENTO`)
        REFERENCES `connectwork`.`TIPO_PAGAMENTO` (`ID_TIPO_PAGAMENTO`),
    FOREIGN KEY (`ID_DADOS_BANCARIOS`)
        REFERENCES `connectwork`.`DADOS_BANCARIOS` (`ID_DADOS_BANCARIOS`)
);
